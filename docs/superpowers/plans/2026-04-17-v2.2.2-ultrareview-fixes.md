def run_coverage():
    """Run coverage command and extract percentage (prefers TOTAL/Total anchor)."""
    if not COVERAGE_COMMAND or COVERAGE_COMMAND == "null":
        return None
    try:
        result = subprocess.run(
            COVERAGE_COMMAND, shell=True, capture_output=True, text=True,
            timeout=TIMEOUT, cwd=PROJECT_ROOT,
        )
        stdout = result.stdout + result.stderr
        # Prefer TOTAL/Total line (pytest-cov, coverage.py, jest, etc.)
        total_line = re.search(r'(?:TOTAL|Total|All files).*?(\d+(?:\.\d+)?)\s*%', stdout)
        if total_line:
            return float(total_line.group(1)) / 100.0
        # Fallback: last % value in output (more likely to be a summary than the first)
        matches = re.findall(r'(\d+(?:\.\d+)?)\s*%', stdout)
        if matches:
            return float(matches[-1]) / 100.0

    except Exception:
